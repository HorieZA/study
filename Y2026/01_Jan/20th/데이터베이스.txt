(SELECT salary FROM edu.salaries WHERE from_date = '1986-06-26' AND emp_no = s.emp_no) AS y1986,
(SELECT salary FROM edu.salaries WHERE from_date = '1987-06-26' AND emp_no = s.emp_no) AS y1987

-- 데이터가 없으면 NULL로 보여짐
SELECT s.emp_no, s.from_date,
(SELECT salary FROM edu.salaries WHERE from_date = '1986-06-26' AND emp_no = s.emp_no) AS y1986,
(SELECT salary FROM edu.salaries WHERE from_date = '1987-06-26' AND emp_no = s.emp_no) AS y1987
FROM edu.salaries AS s WHERE from_date = '1987-06-26';

-- NULL인 데이터는 0으로 처리하기 위하여 IFNULL을 선언
SELECT s.emp_no, s.from_date,
IFNULL((SELECT salary FROM edu.salaries WHERE from_date = '1986-06-26' AND emp_no = s.emp_no), 0) AS y1986,
(SELECT salary FROM edu.salaries WHERE from_date = '1987-06-26' AND emp_no = s.emp_no) AS y1987
FROM edu.salaries AS s WHERE from_date = '1987-06-26';

-- [문제 2] 사원의 사번, 이름, 직책 조회 단, 현재 직책만 조회
SELECT * FROM edu.employees;
SELECT * FROM edu.titles;

-- 이렇게 하는 경우 2번 비교하는 방식이나 마찬가지
SELECT e.emp_no, e.first_name, t.title, t.to_date
FROM edu.employees AS e
INNER JOIN edu.titles AS t
ON (e.emp_no = t.emp_no)
WHERE t.to_date = '9999-01-01'
ORDER BY 1;

-- 아래 방식이 정석임 => on에 넣어야 비교할때 한번에 비교
SELECT e.emp_no, e.first_name, t.title, t.to_date
FROM edu.employees AS e
INNER JOIN edu.titles AS t
ON (e.emp_no = t.emp_no AND t.to_date = '9999-01-01')
ORDER BY 1;

-- [문제 3] 1960년 이후 태어난 사원들의 사원 번호, 부서 번호 조회
SELECT * FROM edu.employees WHERE emp_no = 10001;
SELECT * FROM edu.dept_emp WHERE emp_no = 10001;

-- dept_emp에는 birth_date가 없으므로 이럴땐 WHERE 조건을 사용
SELECT e.emp_no, de.dept_no, e.birth_date
FROM edu.employees AS e
INNER JOIN edu.dept_emp AS de
ON (e.emp_no = de.emp_no)
WHERE e.birth_date >= '1960-01-01'
ORDER BY 3;

-- 우후훗
SELECT e.emp_no, de.dept_no, e.birth_date, SUBSTR(e.birth_date, 1, 4) AS 생일
FROM edu.employees AS e
INNER JOIN edu.dept_emp AS de
ON (e.emp_no = de.emp_no)
WHERE SUBSTR(e.birth_date, 1, 4) >= 1960
ORDER BY 3 DESC;
