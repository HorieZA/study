-- 테이블 생성
CERATE TABLE test(
	`no`		INT			NOT NULL AUTO_INCREMENT PRIMARY KEY,
	`name`		VACHAR(20)	NOT NULL,
	`regDate`	DATETIME	NOT NULL
);

SELECT * FROM edu.test;

-- 등록
INSERT INTO edu.test (`name`, `regDate`) VALUE ('skt', NOW());

-- 수정
UPDATE edu.test SET `name` = 'BTS' WHERE `no` = 3;
-- 아래와 같이 진행하면 name이 전부 BTS로 변경되니 
-- 변경할 항목을 표시할 WHERE 조건을 필수로 달아 주어야함
UPDATE edu.test SET `name` = 'BTS';

-- 삭제
-- 해당 행 삭제
-- 단 프라이머키로 선언된 값은 초기화되지 않음
DELETE FROM edu.test WHERE `no` = 1;

-- DROP
-- 데이터 초기화... 사실상 해당 테이블 삭제 후 CERATE를 이용해 삭제한 테이블 생성
DROP TABLE edu.test;

-- DROP > CERATE
-- 삭제 후 생성 한다
-- 단, 해당 테이블이 기존에 존재하고 있어야함.
TRUNCATE edu.test
